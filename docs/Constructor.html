Example
```
const Horloge = require('horloge');
// Using an async function so we can use await
(async () => {
	const delay = duration =>
		new Promise(resolve => setTimeout(resolve, duration));

	let t = new Horloge();
	t.start();
	await delay(1000);
	t.stop();

	t.log(); // Timer took ~1000 ms
})();
```
